void main() {
    var x = 5;
    var y = 2;

    var addition = x+y;
    var subtraction = x-y;
    var multiplication = x*y;
    var division = x/y;
    var modulus = x%y;

    print('x+y = $addition');
    print('x-y = $subtraction');
    print('x*y = $multiplication');
    print('x/y = $division');
    print('x%y = $modulus');
}
